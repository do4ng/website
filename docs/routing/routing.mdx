zely.js automatically generates routes based on your file tree of pages.

| filename              | result(path)      |
| --------------------- | ----------------- |
| `/index.ts`           | `/`               |
| `/hello.ts`           | `/hello`          |
| `/foo/bar.ts`         | `/foo/bar`        |
| `/user/[id].ts`       | `/user/:id`       |
| `/user/[id]/about.ts` | `/user/:id/about` |
| `/foo/[...bar]`       | `/foo/:bar*`      |

## Route Parameters

You can access the current page parameters by `ctx.params`.

<Tabs>
<TabItem id="/user/[id].ts">

```ts
// pages/user/[id].ts
import type { Context } from 'zely';

export function get(ctx: Context) {
  ctx.send({ id: ctx.params.id });
}
```

</TabItem>
 
 <TabItem id="/list/[...].ts">

```ts
// pages/list/[...id].ts
import type { Context } from 'zely';

export function get(ctx: Context) {
  // ctx.params["*"] === "a/b/c"
  ctx.send({ path: ctx.params['*'].split('/') });
}
```

</TabItem>
</Tabs>
